def self.fetch_fb_graph_user(graph)
  user = graph.get_object 'me'
  picture = graph.get_object 'me', :fields => 'picture,birthday'
  user.merge! picture
end

    converts height in inches value to a string string format is: [feet]‘“

    # File app/models/profile.rb, line 20
def height_str
  unless self[:height]
    return '0'
  else
    temp = self[:height]
    feet = temp / 12
    inch = temp % 12
  end
  "" + feet.to_s + "'" + inch.to_s + "\""
end
